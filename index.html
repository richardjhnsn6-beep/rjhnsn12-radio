let GROQ_KEY = localStorage.getItem('scholar_groq') || ''; function promptForKey() { const key = prompt('First time setup:\nPaste your FREE Groq key (starts with gsk_)\n\nGet one at: console.groq.com/keys'); if (key && key.startsWith('gsk_')) { GROQ_KEY = key; localStorage.setItem('scholar_groq', key); return true; } return false; } async function ask(){ const q = document.getElementById('q').value.trim(); const chat = document.getElementById('chat'); if(!q) return; if(!GROQ_KEY) { if (!promptForKey()) { chat.innerHTML += `

Scholar: You need a free Groq key to chat. Click here to get one

`; return; } } chat.innerHTML += `

You: ${q}

`; document.getElementById('q').value=""; chat.scrollTop = chat.scrollHeight; try { const res = await fetch("https://api.groq.com/openai/v1/chat/completions",{ method:"POST", headers:{ "Authorization":"Bearer "+GROQ_KEY, "Content-Type":"application/json" }, body:JSON.stringify({ model:"llama-3.1-8b-instant", messages:[{role:"user",content:q}], temperature:0.7 }) }); const data = await res.json(); if(data.error) throw new Error(data.error.message); const reply = data.choices[0].message.content; chat.innerHTML += `

Scholar: ${reply}

`; } catch(e) { chat.innerHTML += `

Scholar: Error: ${e.message}. Check your key at console.groq.com

`; GROQ_KEY = ''; localStorage.removeItem('scholar_groq'); } chat.scrollTop = chat.scrollHeight; }
